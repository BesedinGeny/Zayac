	    for (double i = 0; i < a; i+=precision){    
	    	//здесь где то ошибка, нужно минус поставить где то    
	        result += precision*fabs(pow(M_E,-.5*pow(i,2))+pow(M_E,.5*pow((i+precision),2)))/2.0;
